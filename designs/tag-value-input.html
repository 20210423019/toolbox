🔄 数据流 & 逻辑架构 flow
📥 历史值加载流程
1 输入框 onFocus 触发加载 →
2 检查 historyCache[tagId] 缓存 →
3 有缓存 → 直接渲染；无缓存 → 异步请求后端 →
4 返回 { value: string, count: number, lastUsed: string }[] →
5 按 count 降序排序 → 渲染下拉列表 →
6 写入 historyCache[tagId] 缓存 + localStorage ✓
📤 用户操作分支
A 用户输入文字 → 实时过滤历史列表 ↓
B1 选中已有历史值 → onSelect(tagId, value) →
B2 无匹配 + Enter → onNew(tagId, newValue) →
B3 点击 ✕ 清除 → onClear(tagId) →
C 保存后更新 historyCache + 计数递增 ✓
🧩 组件接口定义 (Props)
interface TagValueInputProps {
  tagId: string;              // 当前标签 ID
  tagName: string;            // 标签名称（用于显示）
  tagType: "text" | "path" | "url";  // 标签类型
  value?: string;              // 当前值（受控）
  placeholder?: string;        // 占位符
  onChange: (value: string) => void;  // 值变更回调
  onSave: (value: string) => void;     // 保存回调
  onCancel?: () => void;      // 取消编辑
  historyFetcher?: (tagId: string) => Promise<HistoryItem[]>;  // 自定义加载器
  maxHistory?: number;         // 最多显示历史条数 (默认 20)
  autoFocus?: boolean;         // 自动聚焦
}
📦 数据模型
interface HistoryItem {
  value: string;        // 历史值内容
  count: number;        // 使用次数
  lastUsed: string;     // 最后使用时间 (ISO)
}

interface ComponentState {
  isOpen: boolean;              // 下拉是否打开
  highlightedIndex: number;     // 键盘导航高亮索引
  historyItems: HistoryItem[];  // 历史值列表
  filteredItems: HistoryItem[]; // 过滤后列表
  isLoading: boolean;           // 是否加载中
  inputValue: string;           // 当前输入值
}
